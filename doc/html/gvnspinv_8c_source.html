00074 /* General Vertical Near-Side Perspective inverse equations--mapping 
00075    x,y to lat/long
00076   ----------------------------------------------------------------*/
00077 gvnspinv(x, y, lon, lat)
00078 
00079 double x;                       /* (O) X projection coordinate */
00080 double y;                       /* (O) Y projection coordinate */
00081 double *lon;                    /* (I) Longitude */
00082 double *lat;                    /* (I) Latitude */
00083 
00084 {
00085 double asinz();
00086 double adjust_lon();
00087 double rh;
00088 double r;
00089 double con;
00090 double com;

00093 
00094 /* Inverse equations
00095   -----------------*/
00096 x -= false_easting;
00097 y -= false_northing;
00098 rh = sqrt(x * x + y * y);
00099 r  = rh / R;
00100 con = p - 1.0;
00101 com = p + 1.0;
00102 if (r > sqrt(con/com))
00103    {
00104    p_error("Input data error","gvnsp-for");
00105    return(155);
00106    }
00107 sinz = (p - sqrt(1.0 - (r * r * com) / con)) / (con / r + r/con);
00108 z = asinz(sinz);
00109 sincos(z,&sinz,&cosz);
00110 *lon = lon_center;
00111 if (fabs(rh) <= EPSLN)
00112    {
00113    *lat = lat_center;
00114    return(OK);
00115    }
00116 *lat = asinz(cosz * sin_p15 + ( y * sinz * cos_p15)/rh);
00117 con = fabs(lat_center) - HALF_PI;
00118 if (fabs(con) <= EPSLN)
00119    {
00120    if (lat_center >= 0.0)
00121       {
00122       *lon = adjust_lon(lon_center + atan2(x, -y));
00123       return(OK);
00124       }
00125    else
00126       {
00127       *lon = adjust_lon(lon_center - atan2(-x, y));
00128       return(OK);
00129       }
00130    }
00131 con = cosz - sin_p15 * sin(*lat);
00132 if ((fabs(con) < EPSLN) && (fabs(x) < EPSLN))
00133    return(OK);
00134 *lon  = adjust_lon(lon_center + atan2((x * sinz * cos_p15), (con * rh)));
