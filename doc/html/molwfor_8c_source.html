00073 /* Forward equations
00074   -----------------*/

00076 theta = lat;

00080   -----------------------------------------------------*/

00082    {
00083    delta_theta = -(theta + sin(theta) - con)/ (1.0 + cos(theta));
00084    theta += delta_theta;
