00066   ------------------------------------------------*/
00067 haminv(x, y, lon, lat)
00068 double x;                       /* (O) X projection coordinate */
00069 double y;                       /* (O) Y projection coordinate */

00079 /* Inverse equations
00080   -----------------*/
00081 x -= false_easting;
00082 y -= false_northing;
00083 fac = sqrt(4.0 * R * R - (x * x)/ 4.0 - y * y) / 2.0;
