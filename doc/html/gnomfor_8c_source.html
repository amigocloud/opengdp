00070 /* Gnomonic forward equations--mapping lat,long to x,y
00071   --------------------------------------------------*/
00072 gnomfor(lon, lat, x, y)
00073 double lon;                     /* (I) Longitude */

00075 double *x;                      /* (O) X projection coordinate */

00078 {
00079 double adjust_lon();
00080 double dlon;
00081 double sinphi,cosphi;
00082 double coslon;
00083 double g;
00084 double ksp;

00086 
00087 /* Forward equations
00088   -----------------*/
00089 dlon = adjust_lon(lon - lon_center);
00090 sincos(lat,&sinphi,&cosphi);
00091 coslon = cos(dlon);
00092 g = sin_p13 * sinphi + cos_p13 * cosphi * coslon;
00093 if (g <= 0.0)
00094    {
00095    p_error("Point projects into infinity","gnomfor-conv");

00098 ksp = 1.0 / g;
00099 *x = false_easting + R * ksp * cosphi * sin(dlon);
00100 *y = false_northing + R * ksp * (cos_p13 * sinphi - sin_p13 * cosphi * 
00101                 coslon);
