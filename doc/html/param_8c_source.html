00565     sprintf(msg, "%s\n", USAGE);
00566     LogInfomsg(msg);
00567     FreeParam(this);
00568     return (Param_t *)NULL; 
00569   }
00570 
00571   if (this->output_space_def.proj_num == 31) /* ISINUS => proj_num = 31 */
00572     this->output_space_def.isin_type= SPACE_ISIN_NEST_1;
00573 
00574   /* Copy the projection parameters to orig_proj_param to use the decimal
00575      degree values later (GeoTiff output) */
00576   for (i = 0; i < NPROJ_PARAM; i++) {
00577     this->output_space_def.orig_proj_param[i] =
00578       this->output_space_def.proj_param[i];
00579   }
00580 
00581   /* Convert the output projection parameter lat/long values from decimal
00582      degrees to DMS */
00583   if (!Deg2DMS (this->output_space_def.proj_num,
00584                 this->output_space_def.proj_param)) {
00585     sprintf(msg, "resamp: error converting projection parameters from"
00586             "decimal degrees to DMS\n");
00587     LogInfomsg(msg);
00588     FreeParam(this);
00589     return (Param_t *)NULL; 
00590   }
00591 
00592   /* Use the UL and LR corner points to get the UL corner in output
00593      space and the number of lines/samples in the output image. This must
00594      be done for each SDS, since the pixel size might be different. */
00595   if (!ConvertCorners (this)) {
00596     sprintf(msg, "resamp: error determining UL and lines/samples from "
00597             "the input UL and LR corners\n");
00598     LogInfomsg(msg);
00599     FreeParam(this);
00600     return (Param_t *)NULL; 
00601   }
00602 
00603   /* Make sure the lat/long values are between +-180 and +-90 */
00604   if (this->output_space_def.ul_corner_geo.lat < -90.0 ||
00605       this->output_space_def.ul_corner_geo.lat > 90.0  ||
00606       this->output_space_def.lr_corner_geo.lat < -90.0 ||
00607       this->output_space_def.lr_corner_geo.lat > 90.0  ||
00608       this->output_space_def.ul_corner_geo.lon < -180.0 ||
00609       this->output_space_def.ul_corner_geo.lon > 180.0  ||
00610       this->output_space_def.lr_corner_geo.lon < -180.0 ||
00611       this->output_space_def.lr_corner_geo.lon > 180.0) {
00612     sprintf(msg, "resamp: invalid output lat/lon corners\n");
00613     LogInfomsg(msg);
00614     FreeParam(this);
00615     return (Param_t *)NULL; 
00616   }
00617 
00618   /* MRTSwath will only process swath data */
00619   if (this->input_space_type != SWATH_SPACE) {
00620     sprintf(msg, "resamp: grid or point data detected. MRTSwath will "
00621             "only process swath data\n");
00622     LogInfomsg(msg);
00623     FreeParam(this);
00624     return (Param_t *)NULL; 
00625   }
00626 
00627   return this;
00628 }

00761 int ReadSDS(Param_t *this)
00762 /*
00763 !C******************************************************************************
00764 !Description: 'ReadSDS' reads the input file for available swath SDSs.
00765 
00766 !Input Parameters:
00767  this           'input' data structure; the following fields are input:
00768                    input_file_name
00769 
00770 !Output Parameters:
00771  this           'input' data structure; the following fields are modified:
00772                    input_sds_name_list, input_sds_nbands, input_sds_bands
00773 
00774  (returns)      int: number of SDSs
00775 
00776 !Team Unique Header:
00777 
00778  ! Design Notes:
00779    1. The file is open for HDF access but closed before returning from the
00780       routine.
00781    2. An error status is returned when:
00782        a. duplicating strings is not successful
00783        b. errors occur when opening the input HDF file
00784        c. memory allocation is not successful
00785    3. Error messages are handled with the 'LOG_RETURN_ERROR' macro.
00786 
00787 !END*****************************************************************************/
00788 {
00789   int i, j;
00790   int32 sd_fid, sds_id;        /* file id and sds id */
00791   int32 nsds;                  /* number of SDSs in the file */
00792   int32 nattr;                 /* number of attributes for the SDS */
00793   int32 data_type;             /* data type of the SDS */
00794   char sds_name[MAX_NC_NAME];  /* SDS name */
00795   int32 rank;                  /* rank of the SDS */
00796   int32 dims[MYHDF_MAX_RANK];  /* dimensions of the SDS */
00797 
00798   /* Open file for SD read access */
00799 
00800   sd_fid = SDstart(this->input_file_name, DFACC_RDONLY);
00801   if (sd_fid == HDF_ERROR)
00802     LOG_RETURN_ERROR("opening input file", "ReadSDS", 0);
00803 
00804   /* Get the list of swaths */
00805 
00806   SDfileinfo(sd_fid, &nsds, &nattr);
00807 #ifdef DEBUG
00808   printf("Number of SDSs: %i\n", (int) nsds);
00809 #endif
00810 
00811   /* Loop through the SDSs to get SDS names and info */
00812 
00813   for (i = 0; i < nsds; i++)
00814   {
00815     /* grab the SDS */
00816     sds_id = SDselect(sd_fid, i);
00817     if (sds_id == HDF_ERROR)
00818       LOG_RETURN_ERROR("selecting input SDS", "ReadSDS", 0);
00819 
00820     /* get information for the current SDS */
00821     if (SDgetinfo(sds_id, sds_name, &rank, dims, &data_type, &nattr) ==
00822         HDF_ERROR)
00823       LOG_RETURN_ERROR("getting SDS information", "ReadSDS", 0);
00824 
00825     /* process the swath SDSs */
00826     strcpy(this->input_sds_name_list[i], sds_name);
00827 #ifdef DEBUG
00828     printf("\nSDgetinfo: sds %i, %s, rank = %i, dims =", i,
00829       sds_name, (int) rank);
00830     for (j = 0; j < rank; j++)
00831       printf(" %i", (int) dims[j]);
00832     printf("\n");
00833 #endif
00834 
00835     /* if the rank is 2 then there is only one band to process, otherwise
00836        the first dimension value contains the number of bands in the SDS */
00837     if (rank == 1)
00838       this->input_sds_nbands[i] = 0;
00839     else if (rank == 2)
00840       this->input_sds_nbands[i] = 1;
00841     else
00842     {
00843       /* 3D product. Use the smallest dim value for the bands. */
00844       this->input_sds_nbands[i] = (int)dims[0];
00845       for (j = 0; j < rank; j++)
00846       {
00847         /* find the smallest dimension */
00848         if ((int)dims[j] < this->input_sds_nbands[i])
00849           this->input_sds_nbands[i] = (int)dims[j];
00850 
00851         /* if the smallest dimension is larger than MAX_VAR_DIMS then set
00852            it to MAX_VAR_DIMS. Most likely it won't be processed anyhow. */
00853         if (this->input_sds_nbands[i] > MAX_VAR_DIMS)
00854           this->input_sds_nbands[i] = MAX_VAR_DIMS;
00855       }
00856     }
00857 
00858     if (rank > MYHDF_MAX_RANK) {
00859       SDendaccess(sds_id);
00860       LOG_RETURN_ERROR("sds rank too large", "ReadSDS", false);
00861     }
00862 
00863     /* process all the bands in this SDS */
00864     for (j = 0; j < this->input_sds_nbands[i]; j++)
00865       this->input_sds_bands[i][j] = 1;
00866 
00867     /* close the SDS */
00868     SDendaccess(sds_id);
00869   }
00870 
00871   /* Close the HDF-EOS file */
00872 
00873   SDend(sd_fid);
00874 
00875   return (int)nsds;
00876 }

00878 bool SDSInfo(Param_t *this)
00879 /*
00880 !C******************************************************************************
00881 !Description: 'SDSInfo' accesses the specified SDSs in the input file and
00882   grabs the band information for each SDS.
00883 
00884 !Input Parameters:
00885  this           'input' data structure; the following fields are input:
00886                    input_file_name, input_sds_name_list, input_sds_nbands
00887 
00888 !Output Parameters:
00889  this           'input' data structure; the following fields are modified:
00890                    input_sds_nbands, input_sds_bands
00891 
00892  (returns)      status:
00893                   'true' = okay
00894                   'false' = error return
00895 
00896 !Team Unique Header:
00897 
00898  ! Design Notes:
00899    1. The file is open for HDF access but closed before returning from the
00900       routine.
00901    2. An error status is returned when:
00902        a. duplicating strings is not successful
00903        b. errors occur when opening the input HDF file
00904        c. memory allocation is not successful
00905    3. Error messages are handled with the 'LOG_RETURN_ERROR' macro.
00906 
00907 !END*****************************************************************************/
00908 {
00909   int i, j;
00910   int32 sd_fid, sds_id;        /* file id and sds id */
00911   int32 sds_index;             /* index of the current SDS */
00912   int32 nattr;                 /* number of attributes for the SDS */
00913   int32 data_type;             /* data type of the SDS */
00914   char sds_name[MAX_NC_NAME];  /* SDS name */
00915   int32 rank;                  /* rank of the SDS */
00916   int32 dims[MYHDF_MAX_RANK];  /* dimensions of the SDS */
00917   char errmsg[M_MSG_LEN+1];    /* error message */
00918   char tmpsds[MAX_STR_LEN];    /* temp copy of the SDS name in case we need
00919                                   to remove the '_'s from the SDS name */
00920   int curr_char;               /* current character in the SDS name */
00921 
00922   /* Open file for SD read access */
00923   sd_fid = SDstart(this->input_file_name, DFACC_RDONLY);
00924   if (sd_fid == HDF_ERROR)
00925     LOG_RETURN_ERROR("opening input file", "SDSInfo", false);
00926 
00927   /* loop through the SDSs */
00928   for (i = 0; i < this->num_input_sds; i++)
00929   {
00930     /* try to find the index for the current SDS name */
00931     sds_index = SDnametoindex(sd_fid, this->input_sds_name_list[i]);
00932     if (sds_index == HDF_ERROR) {
00933       /* if running the GUI, SDS names with blank spaces have '_'s placed
00934          for the blank spaces. replace the '_'s with blank spaces and then
00935          try to find the SDS. */
00936       strcpy(tmpsds, this->input_sds_name_list[i]);
00937       for (curr_char = 0; curr_char < (int) strlen (tmpsds) - 1; curr_char++)
00938       {
00939         /* if this character is an underscore then replace with a blank */
00940         if (tmpsds[curr_char] == '_')
00941           tmpsds[curr_char] = ' ';
00942       }
00943 
00944       /* try to find the index for the new SDS name */
00945       sds_index = SDnametoindex(sd_fid, tmpsds);
00946       if (sds_index == HDF_ERROR) {
00947         sprintf(errmsg, "couldn't get sds index for %s or %s",
00948           this->input_sds_name_list[i], tmpsds);
00949         LOG_RETURN_ERROR(errmsg, "SDSInfo", false);
00950       }
00951 
00952       /* copy the correct SDS name into the SDS name list */
00953       strcpy(this->input_sds_name_list[i], tmpsds);
00954     }
00955 
00956     /* get the current SDS */
00957     sds_id = SDselect(sd_fid, sds_index);
00958     if (sds_id == HDF_ERROR)
00959       LOG_RETURN_ERROR("getting sds id", "SDSInfo", false);
00960 
00961     /* get the current SDS information */
00962     if (SDgetinfo(sds_id, sds_name, &rank, dims, &data_type, &nattr) ==
00963         HDF_ERROR) {
00964       SDendaccess(sds_id);
00965       LOG_RETURN_ERROR("getting sds information", "SDSInfo", false);
00966     }
00967 
00968     if (rank > MYHDF_MAX_RANK) {
00969       SDendaccess(sds_id);
00970       LOG_RETURN_ERROR("sds rank too large", "SDSInfo", false);
00971     }
00972 
00973     /* if the user specified bands to be processed for the current
00974        SDS, then fill in the number of bands in the SDS */
00975     if (this->input_sds_nbands[i] != 0)
00976     {
00977        /* if the rank is 2 then there is only one band to process, otherwise
00978           the first dimension value contains the number of bands in the SDS */
00979        if (rank == 2)
00980          this->input_sds_nbands[i] = 1;
00981        else
00982          this->input_sds_nbands[i] = (int)dims[0];
00983     }
00984 
00985     /* otherwise process all the bands in the SDS by default */
00986     else
00987     {
00988        /* if the rank is 2 then there is only one band to process, otherwise
00989           the first dimension value contains the number of bands in the SDS */
00990        if (rank == 2)
00991          this->input_sds_nbands[i] = 1;
00992        else
00993          this->input_sds_nbands[i] = (int)dims[0];
00994 
00995        /* Process all the bands in this SDS */
00996        for (j = 0; j < this->input_sds_nbands[i]; j++)
00997          this->input_sds_bands[i][j] = 1;
00998     }
00999 
01000     /* close the SDS */
01001     SDendaccess(sds_id);
01002   }
01003 
01004   /* close the HDF-EOS file */
01005   SDend(sd_fid);
01006 
01007   return true;
01008 }
