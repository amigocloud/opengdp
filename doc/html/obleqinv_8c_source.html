00107 y -= false_northing;
00108 N = (n / 2.0) * asin(y / (n * R));

00111 x_prime = 2.0 * sin(M);
00112 y_prime = 2.0 * sin(N) * cos(2.0 * M / m) / cos(M);
00113 temp = sqrt(x_prime * x_prime + y_prime * y_prime) / 2.0;
00114 z = 2.0 * asin(temp);
00115 Az = atan2(x_prime, y_prime);
00116 diff_angle = Az - theta;
00117 sincos(diff_angle, &sin_diff_angle, &cos_diff_angle);
00118 sincos(z, &sin_z, &cos_z);

00120 *lon = adjust_lon(lon_center + atan2((sin_z * sin_diff_angle), (cos_lat_o *
