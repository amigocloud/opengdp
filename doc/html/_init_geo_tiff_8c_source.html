00831 /******************************************************************************
00832 
00833 MODULE:  SetGeoTIFFDatum
00834 
00835 PURPOSE:  Set GeoTIFF tags for the datum to be WGS84, only if the output
00836           ellipsoid or projection parameters represent the WGS84 ellipsoid.
00837 
00838 RETURN VALUE:
00839 Type = none
00840 
00841 HISTORY:
00842 Version  Date   Programmer       Code  Reason
00843 -------  -----  ---------------  ----  -------------------------------------
00844          11/03  Gail Schmidt           Original Development
00845 
00846 NOTES:
00847 
00848 ******************************************************************************/
00849 void SetGeoTIFFDatum
00850 (
00851     GeoTIFFFD *geotiff,                 /* I: GeoTIFF descriptor */
00852     Space_def_t *outproj,               /* I:  projection info */
00853     char *citation                      /* I/O: Add datum citation */
00854 )
00855 
00856 {
00857     if( geotiff == NULL ) {
00858        return;
00859     }
00860 
00861     /* If WGS84 ellipsoid, then we can tag the WGS84 datum, otherwise no
00862        datum can be specified. */
00863     if (outproj->sphere == 8 /* WGS84 */ ||
00864         (outproj->proj_param[0] == 6378137.0 &&
00865          outproj->proj_param[1] == 6356752.31414))
00866     {
00867         strcat( citation, "WGS 1984");
00868         GTIFKeySet( geotiff->gtif, GeogGeodeticDatumGeoKey, TYPE_SHORT,
00869                     1, Datum_WGS84 );
00870         GTIFKeySet( geotiff->gtif, GeographicTypeGeoKey, TYPE_SHORT, 1,
00871                     GCS_WGS_84 );
00872     }
00873     else
00874     {
00875         SetGeoTIFFSphere( geotiff, outproj, citation );
00876     }
00877 }

00880 /******************************************************************************
00881 
00882 MODULE:  SetGeoTIFFSphere
00883 
00884 PURPOSE:  Set GeoTIFF tags for the sphere-based projection (no datum)
00885 
00886 RETURN VALUE:
00887 Type = none
00888 
00889 HISTORY:
00890 Version  Date   Programmer       Code  Reason
00891 -------  -----  ---------------  ----  -------------------------------------
00892          11/03  Gail Schmidt           Original Development
00893 
00894 NOTES:
00895 
00896 ******************************************************************************/
00897 void SetGeoTIFFSphere
00898 (
00899     GeoTIFFFD *geotiff,                 /* I:  GeoTIFF descriptor */
00900     Space_def_t *outproj,               /* I:  projection info */
00901     char *citation                      /* I/O:  add datum citation */
00902 )
00903 
00904 {
00905     char tmpstr[256];
00906     double proj_param[2];   /* local copy of first two projection params */
00907 
00908 
00909     /* copy the first two projection parameters */
00910     proj_param[0] = outproj->orig_proj_param[0];
00911     proj_param[1] = outproj->orig_proj_param[1];
00912 
00913     /* if the first two projection parameters weren't specified, then an
00914        ellipsoid was specified. determine the semi-major and semi-minor axis
00915        for the ellipsoid. */
00916     if (proj_param[0] < 1.0 && proj_param[1] < 1.0)
00917     {
00918         proj_param[0] = Proj_sphere[outproj->sphere].major_axis;
00919         proj_param[1] = Proj_sphere[outproj->sphere].minor_axis;
00920     }
00921 
00922     /* use the projection parameters for the semi-major and semi-minor axes */
00923     sprintf( tmpstr, "No Datum. Semi-major axis: %f, "
00924              "Semi-minor axis: %f", proj_param[0], proj_param[1] );
00925     strcat( citation, tmpstr );
00926     GTIFKeySet( geotiff->gtif, GeogGeodeticDatumGeoKey,
00927                 TYPE_SHORT, 1, KvUserDefined );
00928     GTIFKeySet( geotiff->gtif, GeographicTypeGeoKey,
00929                 TYPE_SHORT, 1, KvUserDefined );
00930 
00931     GTIFKeySet( geotiff->gtif, GeogSemiMajorAxisGeoKey,
00932                 TYPE_DOUBLE, 1, proj_param[0] );
00933     if ( proj_param[1] != 0.0 )
00934         GTIFKeySet( geotiff->gtif, GeogSemiMinorAxisGeoKey,
00935                     TYPE_DOUBLE, 1, proj_param[1] );
00936     else
00937         /* If this is a sphere, use the radius for the semi-minor
00938            value as well. */
00939         GTIFKeySet( geotiff->gtif, GeogSemiMinorAxisGeoKey,
00940                     TYPE_DOUBLE, 1, proj_param[0] );
00941 }
