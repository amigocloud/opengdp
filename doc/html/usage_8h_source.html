00070 "     swath2grid - resample one or more SDS from an HDF file \n" \
00071 " \n" \
00072 "SYNOPSIS \n" \
00073 "    swath2grid -if=<input file> -of=<output file> -gf=<geolocation file>\n" \
00074 "           [-off=<output file format (HDF_FMT, GEOTIFF_FMT, RB_FMT)>]\n" \
00075 "           [-sds=<SDS name>] [-kk=<resampling type (NN, BI, CC)>]\n" \
00076 "           -oproj=<output projection>\n" \
00077 "           [-oprm=<output projection parameters>]\n" \
00078 "           [-opsz=<output pixel sizes>]\n" \
00079 "           [-oul=<output upper left corner>]\n" \

00081 "           [-ozn=<output zone number>]\n" \
00082 "           [-osst=<output spatial subset type\n" \
00083 "                   (LAT_LONG, PROJ_COORDS, LINE_SAMPLE)>]\n" \
00084 "           [-osp=<output sphere number>]\n" \
00085 "           [-oty=<output data type>]\n" \
00086 "           [-pf=<parameter file>]\n" \

00088 "DESCRIPTION \n" \
00089 "    Resample one or more SDSs from a L2 MODIS granule into user-specified\n"\
00090 "    projection coordinates.  The required 2D data slice in a 3D SDS is\n" \
00091 "    specified by including the number (0-based) identifying the slice in\n" \
00092 "    the SDS name. i.e. to specify 3rd band in the 3D SDS EV_500_RefSB of a\n"\
00093 "    MOD02HKM granule use -sds=EV_500_RefSB,0,0,1. Multiple SDSs can also\n" \
00094 "    be projected. \n" \
00095 "    i.e. EV_1KM_RefSB,0,1,0,1; EV_500_Aggr1km_RefSB,1; EV_250_Aggr1km_RefSB\n"\
00096 "    Output file types include HDF, GeoTiff, and Raw Binary. Multiple SDSs\n" \
00097 "    will be written to the same HDF file, however separate files will\n" \
00098 "    created for multiple SDSs/bands being output to GeoTiff or Raw Binary\n" \
00099 "    files.  In addition, for multiple output resolutions, SDSs with the\n" \
00100 "    same output resolution will be written to the same HDF file. The output\n"\
00101 "    resolution will be appended to the basename if multiple output\n" \
00102 "    resolutions are specified.\n" \
00103 "    i.e mod02hkm_1000m.hdf, mod02hkm_500m.hdf, mod02hkm_250m.hdf\n" \
00104 " \n" \

00106 "    -help                      Display this help message \n" \
00107 "    -help=proj                 Display general help on map projections \n" \
00108 "                               and spheres \n" \
00109 "    -help=param                Display help on parameter files \n" \
00110 "    -help=projection           Display help for a specific map projection \n" \
00111 "                               (number or short name)\n" \
00112 "    -if=input filename         Input L1 or L2 (swath) granule \n" \
00113 "    -of=output filename        Output filename \n" \
00114 "                               No file extension is needed since the output\n"\
00115 "                               file format will specify the extension.\n" \
00116 "                               If an extension of .hdf, .hdr, or .tif is\n" \
00117 "                               specified, then it will be ignored.\n" \
00118 "    -off=output file format    Output file format (HDF_FMT, GEOTIFF_FMT,\n" \
00119 "                               RB_FMT). Default is HDF_FMT.\n" \
00120 "    -gf=geoloc filename        Associated geolocation file for the input\n" \
00121 "                               file\n" \
00122 "    -sds=SDS name              Name of SDS to resample (use quotes for more\n"\
00123 "                               than one specified SDS)\n" \
00124 "                               Use the index number (0-based) after the SDS\n"\
00125 "                               name to identify 2D slice of 3D SDS. \n" \
00126 "                               e.g. -sds=Cloud_Mask,1 for the first band of\n"\
00127 "                               SDS Cloud_Mask in MOD35 granule. \n" \
00128 "                               e.g. EV_1KM_RefSB,0,1,0,1;\n" \
00129 "                                    EV_500_Aggr1km_RefSB,1;\n" \
00130 "                                    EV_250_Aggr1km_RefSB\n" \
00131 "                               for the second and fourth bands of\n" \
00132 "                               EV_1KM_RefSB, first band of\n" \
00133 "                               EV_500_Aggr1km_RefSB, and all bands of\n" \
00134 "                               EV_250_Aggr1km_RefSB \n" \
00135 "                               Default is to process all swath SDSs. In\n" \
00136 "                               this case, some SDSs may not be of the\n" \
00137 "                               nominal MODIS scan size and therefore an\n" \
00138 "                               error message will be output when the\n" \
00139 "                               MRTSwath tries to process them.\n" \
00140 "    -kk=kernel                 Resampling kernel type (CC, NN, or BI) \n" \
00141 "                               Default is NN.\n" \
00142 "    -oproj=output projection   Output projection number or short name. \n" \
00143 "                               AEA, ER, GEO, GOODE, HAMMER, ISIN, LAMAZ, \n" \
00144 "                               LAMCC, MERCAT, MOLL, PS, SNSOID, TM, and UTM\n"\
00145 "    -oprm=output projection parameters  Output projection parameters (up\n" \
00146 "                               to 15).  Defaults to all zeros.\n" \
00147 "                               Lat/Long values are entered in decimal\n" \
00148 "                               degrees\n" \
00149 "    -opsz=output pixel sizes   Output pixel sizes (up to the number of \n" \
00150 "                               SDSs being processed)  If the number of \n" \
00151 "                               output pixel sizes is less than the \n"\
00152 "                               number of SDSs, then the last pixel size \n" \
00153 "                               is used for the rest of the pixel sizes. \n" \
00154 "                               If the output pixel size is not \n" \
00155 "                               specified, then the resolution of the \n" \
00156 "                               input product will be determined and \n" \
00157 "                               used as the output pixel size. \n" \
00158 "                               Units is meters for all projections except\n" \
00159 "                               GEO, which is degrees.\n" \
00160 "    -oul=output upper left corner  Upper left corner in output space\n" \
00161 "                               (long, lat in decimal degrees)\n" \
00162 "                               (output proj x,y in meters: degrees for Geo)\n"\
00163 "                               (input sample, line - in the case of \n" \
00164 "                               multires products, specify the line/sample \n"\
00165 "                               for the highest resolution SDS to be\n" \
00166 "                               processed) Default is to use the bounding\n" \
00167 "                               coords.\n" \
00168 "    -olr=output lower right corner  Lower right corner in output space\n" \
00169 "                               (long, lat in decimal degrees)\n" \
00170 "                               (output proj x,y in meters: degrees for Geo)\n"\
00171 "                               (input sample, line - in the case of \n" \
00172 "                               multires products, specify the line/sample \n"\
00173 "                               for the highest resolution SDS to be\n" \
00174 "                               processed) Default is to use the bounding\n" \
00175 "                               coords.\n" \
00176 "    -osst=output spatial subset type  Type of output spatial subset info\n" \
00177 "                               (LAT_LONG, PROJ_COORDS, LINE_SAMPLE)\n" \
00178 "                               Default is LAT_LONG.\n" \
00179 "    -ozn=output zone number    Output zone number.  Only needed for UTM.\n" \
00180 "    -osp=output sphere number  Output sphere number.  Not needed if \n" \
00181 "                               first output projection parameter is set.\n" \
00182 "    -oty=output data type      Output data type (CHAR8, UINT8, INT8, \n" \
00183 "                               INT16, UINT16, INT32, UINT32)\n" \
00184 "                               Default is same as input data type.\n" \
00185 "    -pf=parameter file         Parameter file\n" \
00186 "\n" \
00187 "Examples:\n" \
00188 "\n" \
00189 "swath2grid -if=/Modis/testdata/MOD021KM.A2003283.1655.004.2003283232504.hdf\n"\
00190 "  -gf=/Modis/testdata/MOD03.A2003283.1655.004.2003283225954.hdf\n" \
