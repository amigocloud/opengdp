00057 /* Wagner IV forward equations--mapping lat,long to x,y
00058   ----------------------------------------------------*/
00059 wivfor(lon, lat, x, y)
00060 double lon;                     /* (I) Longitude */
00061 double lat;                     /* (I) Latitude */
00062 double *x;                      /* (O) X projection coordinate */
00063 double *y;                      /* (O) Y projection coordinate */
00064 {
00065 double adjust_lon();    /* Function to adjust longitude to -180 - 180 */
00066 double delta_lon;       /* Delta longitude (Given longitude - center */
00067 double theta;
00068 double delta_theta;
00069 double con;
00070 long i;
00071 
00072 /* Forward equations
00073   -----------------*/
00074 delta_lon = adjust_lon(lon - lon_center);
00075 theta = lat;
00076 con = 2.9604205062 * sin(lat);
00077 
00078 /* Iterate using the Newton-Raphson method to find theta
00079   -----------------------------------------------------*/
00080 for (i=0;;i++)
00081    {
00082    delta_theta = -(theta + sin(theta) - con) / (1.0 + cos(theta));
00083    theta += delta_theta;
00084    if (fabs(delta_theta) < EPSLN) break;
00085    if (i >= 30) p_error("Iteration failed to converge","wagneriv-forward");
00086    }
00087 theta /= 2.0;
00088 *x = 0.86310 * R * delta_lon * cos(theta) + false_easting;
00089 *y = 1.56548 * R * sin(theta) + false_northing;
00090 return(OK);
