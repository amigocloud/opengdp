00076 /* General Vertical Near-Side Perspective forward equations--mapping 
00077    lat,long to x,y
00078   ----------------------------------------------------------------*/
00079 gvnspfor(lon, lat, x, y)
00080 double lon;                     /* (I) Longitude */
00081 double lat;                     /* (I) Latitude */
00082 double *x;                      /* (O) X projection coordinate */
00083 double *y;                      /* (O) Y projection coordinate */
00084 
00085 {
00086 double adjust_lon();
00087 double dlon;
00088 double sinphi,cosphi;
00089 double coslon;
00090 double g;
00091 double ksp;
00092 
00093 /* Forward equations
00094   -----------------*/
00095 dlon = adjust_lon(lon - lon_center);
00096 sincos(lat,&sinphi,&cosphi);
00097 coslon = cos(dlon);
00098 g = sin_p15 * sinphi + cos_p15 * cosphi * coslon;
00099 if (g < (1.0/ p))
00100    {
00101    p_error("Point cannot be projected","gvnsp-for");
00102    return(153);
00103    }
00104 ksp = (p - 1.0)/(p - g);
00105 *x = false_easting + R * ksp * cosphi * sin(dlon);
00106 *y = false_northing + R * ksp * (cos_p15 * sinphi - sin_p15 * cosphi * coslon);
00107 
00108 return(OK);
00109 }
