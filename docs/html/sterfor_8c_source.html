00069   ---------------------------------------------------*/
00070 sterfor(lon, lat, x, y)
00071 double lon;                     /* (I) Longitude                */
00072 double lat;                     /* (I) Latitude                 */
00073 double *x;                      /* (O) X projection coordinate  */
00074 double *y;                      /* (O) Y projection coordinate  */
00075 {
00076 double adjust_lon();    /* Function to adjust longitude to -180 - 180   */
00077 double sinphi, cosphi;  /* sin and cos value                            */
00078 double dlon;            /* delta longitude value                        */
00079 double coslon;          /* cos of longitude                             */
00080 double ksp;             /* scale factor                                 */
00081 double g;

00083 /* Forward equations
00084   -----------------*/
00085 dlon = adjust_lon(lon - lon_center);
00086 sincos(lat,&sinphi,&cosphi);
00087 coslon = cos(dlon);
00088 g = sin_p10 * sinphi + cos_p10 * cosphi * coslon;
00089 if (fabs(g + 1.0) <= EPSLN)
00090    {
00091    p_error("Point projects into infinity","ster-for");
00092    return(103);
00093    }
00094 else
00095    {
00096    ksp = 2.0 / (1.0 + g);
00097    *x = false_easting + r_major * ksp * cosphi * sin(dlon);
00098    *y = false_northing + r_major * ksp * (cos_p10 * sinphi - sin_p10 * 
00099                         cosphi * coslon);
00100    }
